# Supprimer les anciennes migrations accounts si elles existent
Remove-Item -Recurse -Force apps\accounts\migrations\

# Recréer le dossier migrations

New-Item -ItemType File -Path apps\accounts\migrations\__init__.py

# Créer les migrations
python manage.py makemigrations accounts

# Appliquer les migrations
python manage.py migrate
